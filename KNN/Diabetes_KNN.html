for column in zero_not_accepted:
    diabetes_df[column] = diabetes_df[column].replace(0, np.NaN)#1st Replace 0 with NaN
    mean = int(diabetes_df[column].mean(skipna=True))#calculate mean
    diabetes_df[column] = diabetes_df[column].replace(np.NaN, mean)#then replace NaN with Mean
